CREATE DATABASE PROJETOPONTO;

CREATE TABLE ADMINISTRADOR(
ID_FUNCIONARIO INT PRIMARY KEY AUTO_INCREMENT,
LOGIN_ADMINISTRADOR VARCHAR (100) NOT NULL UNIQUE,
SENHA_ADMINISTRADOR VARCHAR (100) NOT NULL UNIQUE,
DATA_CADASTRO DATE

);

CREATE TABLE COLABORADOR(
ID_COLABORADOR INT PRIMARY KEY AUTO_INCREMENT,
NOME VARCHAR (100) NOT NULL,
CPF VARCHAR (14) NOT NULL UNIQUE,
DEPARTAMENTO VARCHAR (50),
SALARIO DECIMAL (10,2),
DATA_ADMISSAO DATE NOT NULL,
LOGIN_COLABORADOR VARCHAR (100) NOT NULL UNIQUE,
SENHA_COLABORADOR VARCHAR (100) NOT NULL UNIQUE
);

CREATE TABLE REGISTRO_PONTO(
ID_REGISTRO_PONTO INT PRIMARY KEY AUTO_INCREMENT,
COLABORADOR_ID INT,
DATA_HORA_ENTRADA DATE NOT NULL,
DATA_HORA_SAIDA DATE NOT NULL
);
